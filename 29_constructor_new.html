let person = {} person.name = 'suji'; // 프라퍼티(속성,변수) person.introduce = function() { // 프라퍼티에 함수가 담겨 있으면 = 메서드 return 'My name is ' + this.name; } console.log(person.introduce()); // My name is suji let person = { 'name' : 'suji', 'introduce' : function() { return 'My name is ' + this.name; } } console.log(person.introduce()); // My name is suji let person1 = { 'name' : 'suji', 'introduce' : function() { // 메소드 중복 return 'My name is ' + this.name; } } console.log(person1.introduce()); // My name is suji let person2 = { 'name' : 'lucky', 'introduce' : function() { // 메소드 중복 return 'My name is ' + this.name; } } console.log(person2.introduce()); // My name is lucky
