let a = 1; let b = a; // 복제 b = 2; console.log(a) // 1 console.log(b) // 2 let a = {'id':1}; let b = a; b.id= 2; console.log(a.id); // 2 let a = {'id':1}; let b = a; b = {'id':2}; // 새로운 객체를 생성해서 할당함. 따라서, b는 a가 생성한 객체를 바라보는 것이 아니라, 새롭게 생성한 별개의 객체를 바라보게 되는 것임! console.log(a.id); // 1 객체의 경우, 각각 변수가 똑같은 객체를 바라보게 되는 것이고, 다른 객체가 할당되면 바라보는 객체가 달라지게 되는 것이지. let a = 1; // a = 1 function func(b) { // b = a // b = 1 b = 2; // b = 2 console.log(b) } func(a); console.log(a); // 1 let a = {'id':1}; function func(b){ // a = b // b = {'id':1} b = {'id':2}; // b = {'id':2} } func(a); console.log(a.id); // 1 let a = {'id':1}; function func(b){ // a = b // b = {'id':1} b.id = 2; // b = {'id':2} // a = {id:2} } func(a); console.log(a.id); // 2
